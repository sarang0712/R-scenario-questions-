is_palindrome <- function(x) {
  x <- as.character(x)                 # Convert to character
  reversed <- paste(rev(strsplit(x, "")[[1]]), collapse = "")
  return(x == reversed)                # Compare original and reversed
}

# Examples
is_palindrome("madam")
is_palindrome(121)
is_palindrome("hello")
